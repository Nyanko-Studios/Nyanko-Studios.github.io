NMI::

			LDA LOGIC_DONE

			PHA

			STA MMC3_IRQ_DISABLE

			STA MMC3_IRQ_RELOAD								; reload IRQ
			STA MMC3_IRQ_ENABLE								; enable IRQ

				LDX #$00									; count from 0 to data @ $0102
-:
					LDA $0103,X								; write buffer data to PPU

					INX
					CPX $0102								; repeat while X != $0102

				CLC
				ADC $10
				STA $10
				
				LDA offsetaddhi - 1,X
				ADC $11
				STA $11
				
				LDY #$00
-:
					LDA ($10),Y
					STA $0103,Y
					INY
					CPY #size
					BNE -
				INC drawframe
				DEC loadframe
				
				
				; prep text drawing
				LDX #$00
				STX textPos
				STX textRow
				
				LDX #$03
				STX clearTextArea
				
+:
			RTS


LOADTEXTSTRING::
			LDA loadtext
			BNE #$03
				JMP +
				
				LDA updatetimer			; wait for timer
				BEQ #$03
					JMP dectimertextstr
				
				LDY animationFrame
				LDA storylo,Y
				STA $10
				LDA storyhi,Y
				STA $11
				
				LDX textRow
				BEQ ++
					LDA storyrowadd,Y
					STA $12
-:
						LDA $12
						CLC
						ADC $10
						STA $10
						
						LDA #$00
						ADC $11
						STA $11
						
						DEX
						BNE -
++:
				
				; set timer
				
				LDX textRow
				
				LDY textPos
				LDA ($10),Y
				STA $0103
				BNE ++
					STA updatetimer
					CPX #$02
					BNE #$02
						DEC loadtext
						;DEC drawframe
						; set long timer
					INC textRow
					LDA #$FF
					STA textPos
++:
				LDA #$01				; buffer length
				STA $0102
				
				;LDX textRow
				
				LDA offsettextrowlo,X
				STA $0100
				
				LDA offsettextrowhi,X
				STA $0101
				
				; special frames 
				LDA animationFrame
				CMP #$04
				BNE ++
					LDA #$0F
					CMP textPos
					;BCS #$04			; wait longer if at ellipsis
					;	LDA #$28		; temp delay time for timer until actual delay time found
					;	STA updatetimer
					LDX textRow								; if passed row 0 & on animationFrame 4
					BEQ ++									; end drawing text
						DEC loadtext
						LDA #$03
						STA textRow
++:
				;LDA animationFrame
				CMP #$05
				BNE ++
					LDA #$22
					STA $0101
					LDX textRow
					BEQ @case0								; row 0
					CPX #$01
					BEQ @case1								; row 1
					BNE @case2								; row 2
@case0:
						LDA #$8B
						STA $0100
						BNE ++
@case1:
						LDA #$CE
						STA $0100
						BNE ++
@case2:
						DEC loadtext							; don't draw
						LDA #$03
						STA textRow
++:
				
				LDA $0100
				CLC
				ADC textPos
				STA $0100
				
				LDX animationFrame
				
				LDA $0103		; get char to write
				BEQ @case0		; NULL "\0"
				CMP #$20
				BEQ @case1		; space " "
				CMP #$2C
				BEQ @case2		; comma ","
				CMP #$2E
				BEQ @case3		; period "."
				CMP #$3A
				BEQ @case4		; colon ":"
				CMP #$7A
				BCC @case5		; any char < #$7A "z"
				BRK #$03
@case0:
					LDA #$01			; #$00
					BNE ++
@case1:
					LDA #$02			; " "
					BNE ++
@case2:
					LDA #$10			; ","
					BNE ++
@case3:
					LDA #$20			; "."
					CPX #$05			; if animationFrame = 5, don't wait longer
						BEQ @case5
					BNE ++
@case4:
					LDA #$10			; ":"
					BNE ++
@case5:
					LDA #$02			; other
					BNE ++
++:
				STA updatetimer			; delay values not final
				
				INC textPos
				INC drawframe
				CLC
				BCC +
				;INC loadframe
dectimertextstr:
				DEC updatetimer
+:
				
			RTS
			
CLEARTEXT::
			LDX clearTextArea
			BEQ +
				LDA offsettextrowlo - 1,X
				STA $0100
				LDA offsettextrowhi - 1,X
				STA $0101
				LDX #$20
				STX $0102
				LDA #$00
-:
					STA $0103 - 1,X
					DEX
					BNE -
				INC drawframe
				DEC clearTextArea
+:
			RTS
			
FADEFRAME::
			LDA fadetimer
			BNE #$01
				RTS
			
			LDA unfadeflag
			BNE +
				DEC fadetimer
				BNE ++
				BEQ ++
+:
				INC fadetimer
++:
			
			LDA fadetimer
			CMP #25
			BEQ @case6
			CMP #20
			BEQ @case1
			;CMP #$28
			;BEQ @case0
			CMP #15
			BEQ @case2
			;CMP #$18
			;BEQ @case0
			CMP #10
			BEQ @case3
			CMP #5
			BEQ @case4
			CMP #$00
			BEQ @case5
			BNE @default
@case6:
				LDA unfadeflag
				BEQ @case0
					LDA #$00
					STA fadetimer
					INC loadtext
					DEC unfadeflag
					LDX #$00					; fadepalette index
					BEQ @wrpal					; restore palette
				
@case0:
				LDA pPPUMASK					; set color emphasis
				;EOR #$E0						; EOR for testing
				BNE @default					; just RTS instead ?
				STA pPPUMASK
				BRK #$82
@case1:
				LDX #5							; index into fadepalette
				BNE @wrpal
@case2:
				LDX #10
				BNE @wrpal
@case3:
				LDX #15
				; BNE @wrpal
@wrpal:
				LDA pPPUMASK					; clear color emphasis
				AND #$1F
				STA pPPUMASK
				LDA #$01
				STA $0100						; set write address
				LDA #$3F
				STA $0101
				LDA #$05						; set buffer length
				STA $0102
				LDA fadepalette,X				; write palette to buffer
				STA $0103
				INX
				LDA fadepalette,X
				STA $0104
				INX
				LDA fadepalette,X
				STA $0105
				INX
				LDA fadepalette,X
				STA $0106
				INX
				LDA fadepalette,X
				STA $0107
				INC drawframe
				BNE @default					; just RTS instead ?
				BRK #$80
@case4:
				LDA unfadeflag
				BNE @enbkgd
					LDA pPPUMASK				; turn off the background
					AND #%11110111
					STA pPPUMASK
					BNE @default				; just RTS instead ?
@enbkgd:
					LDA pPPUMASK				; enable background
					ORA #$08
					STA pPPUMASK
					BNE @default
				BRK #$81
@case5:
				INC updateframe					; render next story frame
				INC fadetimer					; make not 0 so unfade will occur
				INC unfadeflag					; count up
@default:
			
			RTS
			
PRG_CHECKSUM::
			RTS
			
CHR_CHECKSUM::
			LDA #$00
			STA $20			; sum LO
			STA $21			; sum HI
			TAX				; address LO
			STA $23			; address HI
			TAY				; bank
			
-:
					;LDA #$00			; slot 0
					STX MMC3_BANK_SEL
					LDA $24				; select bank
					STY MMC3_BANK_DATA
					BIT PPUSTATUS
					;LDA #$00			; set address
					STX PPUADDR
					STX PPUADDR
					LDA PPUDATA			; clear PPU buffer
					
--:
					LDA PPUDATA			; 16-bit add
					CLC
					ADC $20
					STA $20
					
					LDA #$00
					ADC $21
					STA $21
					
					INX
					BNE #$02
						INC $23
					
					LDA $23
					CMP #$08
					BNE --
				
				TXA			; A = $00
				STA $23
				
				INY
				INY
				CPY #$44			; don't need to waste time checking all empty banks
				BNE -
			
			LDA $20
			CMP $FFF2
			BEQ #$02
				BRK #$07
			LDA $21
			CMP $FFF3
			BEQ #$02
				BRK #$08
			
			RTS
